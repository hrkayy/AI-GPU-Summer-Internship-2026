from diffusers import StableDiffusionPipeline
import torch
 
# ─────────────────────────────────────────────────────────────────────────────
# STEP 1: LOAD THE MODEL
# ─────────────────────────────────────────────────────────────────────────────
# StableDiffusionPipeline automatically loads ALL components:
#   - VAE Encoder + Decoder
#   - CLIP Text Encoder
#   - U-Net Denoiser
#   - Noise Scheduler (controls the denoising steps)
 
pipe = StableDiffusionPipeline.from_pretrained(
    "stabilityai/stable-diffusion-2-1",   # The model name on HuggingFace Hub
    torch_dtype = torch.float16,           # Use 16-bit floats (FP16)
    # Why FP16? The H200 has special hardware for FP16 (Tensor Cores)
    # FP16 uses half the memory of FP32 and runs ~3x faster
    # The quality difference is negligible for image generation
    variant = "fp16",
    use_safetensors = True,
)
 
# Move ALL model components to GPU (H200)
pipe = pipe.to("cuda")
 
# ─────────────────────────────────────────────────────────────────────────────
# STEP 2: ENABLE MEMORY OPTIMISATIONS
# ─────────────────────────────────────────────────────────────────────────────
 
# xFormers: a library that implements memory-efficient attention operations
# The standard attention mechanism uses O(n²) memory — xFormers reduces this
# This alone can save 30-40% VRAM and speed up generation by 20-30%
pipe.enable_xformers_memory_efficient_attention()
 
# VAE slicing: if generating multiple images, process VAE one image at a time
# Reduces VRAM spikes at the expense of very slightly slower speed
pipe.enable_vae_slicing()
 
# ─────────────────────────────────────────────────────────────────────────────
# STEP 3: GENERATE AN IMAGE
# ─────────────────────────────────────────────────────────────────────────────
 
prompt = "A majestic snow leopard on a mountain peak at sunrise, photorealistic, 4K"
 
negative_prompt = "blurry, low quality, distorted, watermark, cartoon, text overlay"
# negative_prompt = what you DON'T want in the image
# Common negative prompts: "blurry, low quality, ugly, oversaturated"
 
# torch.autocast enables Automatic Mixed Precision (AMP)
# AMP uses FP16 for most operations but FP32 for numerically sensitive ones
with torch.autocast("cuda"):
    result = pipe(
        prompt              = prompt,
        negative_prompt     = negative_prompt,
        height              = 512,                  # Output image height
        width               = 512,                  # Output image width
        num_inference_steps = 30,                   # Number of denoising steps
        # More steps = sharper/more detailed, but slower
        # 20 steps: fast (2s), decent quality
        # 30 steps: good balance (3s) ← use this as default
        # 50 steps: best quality (5s), diminishing returns after this
        guidance_scale      = 7.5,
        # guidance_scale controls how closely to follow the prompt
        # Low  (1-3):  creative but may not match prompt
        # Med  (7-10): good balance ← default
        # High (12+):  very literal, can look over-saturated
        num_images_per_prompt = 1,
        generator = torch.Generator("cuda").manual_seed(42)
        # Setting a seed makes the output reproducible — same seed = same image
        # Change the seed to get different images from the same prompt
    )
 
image = result.images[0]   # Get the first (and only) generated image
image.save("snow_leopard.png")
print(f"Image saved! Size: {image.size}")   # Image saved! Size: (512, 512)



# Experiment: Generate the same prompt with different guidance scales
# This helps you understand what guidance_scale actually does
 
prompt = "A futuristic city with flying cars, cyberpunk style, neon lights"
 
guidance_values = [1.0, 3.0, 7.5, 12.0, 15.0]   # From very free to very strict
images = []
 
for gs in guidance_values:
    with torch.autocast("cuda"):
        result = pipe(
            prompt              = prompt,
            num_inference_steps = 30,
            guidance_scale      = gs,
            generator = torch.Generator("cuda").manual_seed(42)  # Same seed = fair comparison
        )
    images.append(result.images[0])
    print(f"guidance_scale={gs:5.1f}: image generated")
 
# Save all 5 images in a row for comparison
import matplotlib.pyplot as plt
fig, axes = plt.subplots(1, 5, figsize=(20, 4))
for ax, img, gs in zip(axes, images, guidance_values):
    ax.imshow(img)
    ax.set_title(f"guidance={gs}", fontsize=11)
    ax.axis("off")
plt.tight_layout()
plt.savefig("guidance_comparison.png", dpi=120)
print("Saved: guidance_comparison.png")
 
# What to observe:
# guidance=1.0  -> Creative but vague, doesn't look like a cyberpunk city
# guidance=3.0  -> Better, some neon lights visible
# guidance=7.5  -> Clear cyberpunk city with flying cars (best balance)
# guidance=12.0 -> Very detailed but slightly oversaturated / 'too much'
# guidance=15.0 -> Over-processed, colours clipping, artifacts possible


import time
import torch
 
def benchmark_generation(pipe, prompt, steps=30, num_runs=5):
    """
    Measures how fast Stable Diffusion runs on the H200 GPU.
    Reports: average time per image, images per second, VRAM usage.
    """
    # Warmup: first run always slower (CUDA initialization)
    with torch.autocast("cuda"):
        _ = pipe(prompt, num_inference_steps=5)
    torch.cuda.synchronize()   # Wait for GPU to finish before timing
 
    times = []
    for i in range(num_runs):
        torch.cuda.synchronize()        # Ensure GPU is ready
        start = time.perf_counter()     # Start timer (high resolution)
 
        with torch.autocast("cuda"):
            _ = pipe(prompt, num_inference_steps=steps)
 
        torch.cuda.synchronize()        # Wait for GPU to finish
        elapsed = time.perf_counter() - start
        times.append(elapsed)
        print(f"  Run {i+1}: {elapsed:.2f} seconds")
 
    avg_time = sum(times) / len(times)
    imgs_per_sec = 1.0 / avg_time
 
    print(f"
Results for {steps} inference steps:")
    print(f"  Average time per image : {avg_time:.2f} seconds")
    print(f"  Images per second      : {imgs_per_sec:.3f}")
 
    # VRAM usage report
    allocated_gb = torch.cuda.memory_allocated(0) / 1e9
    reserved_gb  = torch.cuda.memory_reserved(0)  / 1e9
    total_gb     = torch.cuda.get_device_properties(0).total_memory / 1e9
 
    print(f"
VRAM Usage:")
    print(f"  Allocated : {allocated_gb:.1f} GB  (actively used)")
    print(f"  Reserved  : {reserved_gb:.1f} GB  (held by PyTorch)")
    print(f"  Total     : {total_gb:.1f} GB  (H200 has 80GB)")
    print(f"  Free      : {total_gb - reserved_gb:.1f} GB")
 
test_prompt = "A golden retriever puppy playing in autumn leaves, bokeh background"
benchmark_generation(pipe, test_prompt, steps=30, num_runs=5)
